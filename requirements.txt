# =============================================================================
# EXTERNAL SERVICE APIs
# =============================================================================
# These are third-party services the workflows call via HTTP.
# Credentials for each are configured in n8n (Settings → Credentials).
# No SDK installation is required — all calls use the httpRequest node
# or built-in n8n integration nodes.


# --- OpenAI -------------------------------------------------------------------
# API:        https://api.openai.com/v1/chat/completions
# Auth:       API key (Bearer token)
# Models used:
#   gpt-4o        — Orchestrator, Risk Agent, Reporting Agent
#   gpt-4o-mini   — Status Agent, Comms Agent
# Credential type in n8n: openAiApi
# n8n credential name:    OpenAi account


# --- Jira Software Cloud ------------------------------------------------------
# API:        Atlassian REST API v2/v3
# Auth:       Basic auth (email + API token)
# Permissions required:
#   - Browse projects
#   - Create issues
#   - Read sprint data
# Credential type in n8n: jiraSoftwareCloudApi
# n8n credential name:    Jira SW Cloud account
# Token URL:  https://id.atlassian.com/manage-profile/security/api-tokens


# --- Airtable -----------------------------------------------------------------
# API:        https://api.airtable.com/v0/
# Auth:       Personal Access Token
# Scopes required:
#   data.records:read
#   data.records:write
#   schema.bases:read
# Tables used:
#   Projects          — read by Program Management Workflow
#   Dependency Mapping — read by Risk Agent
# Credential type in n8n: airtableTokenApi
# n8n credential name:    Airtable Personal Access Token account
# Token URL:  https://airtable.com/create/tokens


# --- Google Drive -------------------------------------------------------------
# API:        Google Drive API v3
# Auth:       OAuth2
# Scopes required:
#   https://www.googleapis.com/auth/drive
#   https://www.googleapis.com/auth/drive.file
# Used for:
#   Comms Agent     — search and download .txt meeting notes
#   Reporting Agent — create exec brief document
# Credential type in n8n: googleDriveOAuth2Api
# n8n credential name:    Google Drive account


# --- Gmail --------------------------------------------------------------------
# API:        Gmail API v1
# Auth:       OAuth2 (same Google account as Drive)
# Scopes required:
#   https://www.googleapis.com/auth/gmail.send
# Used for:
#   Slack Approval Handler — send approved brief to exec recipient
# Credential type in n8n: gmailOAuth2
# n8n credential name:    Gmail account


# --- Slack --------------------------------------------------------------------
# API:        Slack Web API (chat.postMessage)
#             Slack Interactivity (webhook for button payloads)
# Auth:       OAuth2 Bot Token
# Scopes required:
#   chat:write
#   chat:write.public
#   channels:read
# Used for:
#   Program Management Workflow  — post brief to exec channel
#   Slack Approval Handler       — receive button payloads, post confirmations
#   Error Notification           — post error alerts
# Credential type in n8n: slackOAuth2Api
# n8n credential name:    Slack account 2
# App setup:  https://api.slack.com/apps
